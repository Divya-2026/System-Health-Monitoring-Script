import psutil
import logging
from datetime import datetime

# Setup logging
logging.basicConfig(filename='system_health.log', level=logging.INFO,
                    format='%(asctime)s - %(levelname)s - %(message)s')

# Thresholds
CPU_THRESHOLD = 80  # %
MEMORY_THRESHOLD = 80  # %
DISK_THRESHOLD = 90  # %

def check_cpu():
    cpu = psutil.cpu_percent(interval=1)
    if cpu > CPU_THRESHOLD:
        logging.warning(f"High CPU Usage: {cpu}%")
    return cpu

def check_memory():
    mem = psutil.virtual_memory()
    if mem.percent > MEMORY_THRESHOLD:
        logging.warning(f"High Memory Usage: {mem.percent}%")
    return mem.percent

def check_disk():
    disk = psutil.disk_usage('/')
    if disk.percent > DISK_THRESHOLD:
        logging.warning(f"High Disk Usage: {disk.percent}%")
    return disk.percent

def list_processes():
    processes = []
    for proc in psutil.process_iter(['pid', 'name', 'cpu_percent']):
        processes.append(proc.info)
    return processes

def main():
    cpu = check_cpu()
    memory = check_memory()
    disk = check_disk()
    processes = list_processes()

    print(f"CPU Usage: {cpu}%")
    print(f"Memory Usage: {memory}%")
    print(f"Disk Usage: {disk}%")
    print(f"Top 5 CPU-consuming processes:")
    for p in sorted(processes, key=lambda x: x['cpu_percent'], reverse=True)[:5]:
        print(f"{p['name']} (PID {p['pid']}): {p['cpu_percent']}% CPU")

if __name__ == "__main__":
    main()
